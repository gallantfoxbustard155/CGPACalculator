function addSubject() {
    let div = document.createElement("div");
    div.className = "row";

    div.innerHTML = `
        Grade: <input type="number" step="0.1" class="grade">
        Credit: <input type="number" class="credit">
    `;

    document.getElementById("subjects").appendChild(div);
}

function calculateCGPA() {
    let grades = document.getElementsByClassName("grade");
    let credits = document.getElementsByClassName("credit");

    let totalPoints = 0;
    let totalCredits = 0;

    for (let i = 0; i < grades.length; i++) {
        let g = parseFloat(grades[i].value);
        let c = parseFloat(credits[i].value);

        if (!isNaN(g) && !isNaN(c)) {
            totalPoints += g * c;
            totalCredits += c;
        }
    }

    if (totalCredits === 0) {
        document.getElementById("result").innerText = "Enter valid data!";
        return;
    }

    let cgpa = totalPoints / totalCredits;
    document.getElementById("result").innerText = "CGPA: " + cgpa.toFixed(2);
}
